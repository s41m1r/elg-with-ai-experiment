```sql
SELECT
    visit_occurrence.visit_occurrence_id AS case_id,
    'Medication Administration' AS activity,
    drug_exposure.drug_exposure_start_datetime AS timestamp
FROM
    visit_occurrence
JOIN
    drug_exposure ON visit_occurrence.person_id = drug_exposure.person_id
WHERE
    visit_occurrence.visit_start_datetime <= drug_exposure.drug_exposure_start_datetime
    AND visit_occurrence.visit_end_datetime >= drug_exposure.drug_exposure_start_datetime
    AND drug_exposure.drug_type_concept_id IN (
        SELECT concept_id
        FROM concept
        WHERE concept_class_id = 'Drug Administration'
    );
```
